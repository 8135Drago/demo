import { useState, useEffect } from "react";
import api from "../utils/api";
import { toast, ToastContainer } from "react-toastify";
import "react-toastify/dist/ReactToastify.css";

export default function DownloadPage() {
  const [servers, setServers] = useState([]);
  const [selectedServer, setSelectedServer] = useState("");
  const [formData, setFormData] = useState({
    ip: "",
    username: "",
    port: 22,
    remotePath: "",
  });

  useEffect(() => {
    loadServers();
  }, []);

  const loadServers = async () => {
    try {
      const res = await api.get("/servers");
      setServers(res.data);
    } catch {
      toast.error("Failed to load servers.");
    }
  };

  const handleServerChange = (e) => {
    const ip = e.target.value;
    setSelectedServer(ip);
    const server = servers.find((s) => s.ip === ip);
    if (server) {
      setFormData({
        ...formData,
        ip: server.ip,
        username: server.username,
        port: server.port,
      });
    }
  };

  const handleDownload = async (e) => {
    e.preventDefault();

    if (!formData.remotePath) {
      toast.error("Please enter a remote file path.");
      return;
    }

    try {
      const res = await api.post("/download", formData, {
        responseType: "blob",
      });

      // Download file locally
      const blob = new Blob([res.data]);
      const url = window.URL.createObjectURL(blob);
      const a = document.createElement("a");
      a.href = url;
      a.download = formData.remotePath.split("/").pop();
      document.body.appendChild(a);
      a.click();
      a.remove();
      toast.success("File downloaded successfully!");
    } catch (err) {
      toast.error("Download failed: " + err.message);
    }
  };

  return (
    <div style={styles.container}>
      <h2>File Download</h2>

      <form onSubmit={handleDownload} style={styles.form}>
        <label>Server:</label>
        <select value={selectedServer} onChange={handleServerChange} required>
          <option value="">Select Server</option>
          {servers.map((s) => (
            <option key={s.id} value={s.ip}>
              {s.ip} ({s.username})
            </option>
          ))}
        </select>

        <label>Username:</label>
        <input type="text" value={formData.username} readOnly />

        <label>Port:</label>
        <input type="number" value={formData.port} readOnly />

        <label>Remote File Path:</label>
        <input
          type="text"
          placeholder="/remote/directory/file.txt"
          value={formData.remotePath}
          onChange={(e) =>
            setFormData({ ...formData, remotePath: e.target.value })
          }
          required
        />

        <button type="submit">Download File</button>
      </form>

      <ToastContainer />
    </div>
  );
}

const styles = {
  container: {
    padding: "20px",
    maxWidth: "400px",
    margin: "auto",
    display: "flex",
    flexDirection: "column",
    alignItems: "center",
  },
  form: {
    display: "flex",
    flexDirection: "column",
    gap: "10px",
    width: "100%",
  },
};
